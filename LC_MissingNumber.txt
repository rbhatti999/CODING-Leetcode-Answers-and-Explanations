nums.sort()                     # sort the list of numbers to make it easier to find missing number
a = len(nums)                   # take length of list

if (nums[0] != 0):              # if 0 not in 0 index after sorting list, 0 is missing number
    return 0

elif (nums[-1] != a):           # if a is not in the final index then return a
    return a

else:                           # iterate thru rest of numbers between 0 and a and return missing number
    for i in range(1, a):
        if (nums[i] != i):
            return i
