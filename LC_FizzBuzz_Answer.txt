    ans = []          # initialize our answer list

    for i in range(1, n+1):     # use for loop to check numbers 1 up to n (use n+1 to include n in loop)

        if (i % 5 == 0 and i % 3 == 0):    # check both conditions at same time first to ensure i divisible by both
            ans.append("FizzBuzz")

        elif (i % 5 == 0):                # check i divisible by 5
            ans.append("Buzz")

        elif (i % 3 == 0):                # check i divisible by 3
            ans.append("Fizz")

        else:                             # append i itself if not divided by 3 and/or 5
            ans.append(i)

    return ans
